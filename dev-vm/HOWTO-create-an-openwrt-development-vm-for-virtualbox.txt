# HOWTO Create an OpenWRT development VM for VirtualBox

# DOC https://wiki.openwrt.org/doc/howto/virtualbox
# DOC http://hoverbear.org/2014/11/23/openwrt-in-virtualbox/

### Config
URL="https://downloads.openwrt.org/chaos_calmer/15.05/x86/generic/openwrt-15.05-x86-generic-combined-ext4.img.gz"
VDI="./openwrt.15.05.vdi"
VMNAME="OpenWRT"
SIZE='512000000'

### Creating VDI
curl $URL \
| gunzip \
| VBoxManage convertfromraw --format VDI stdin $VDI $SIZE

### Setup the VM

VBoxManage createvm --name $VMNAME --register && \
VBoxManage modifyvm $VMNAME \
    --description "$(cat HOWTO-create-an-openwrt-development-vm-for-virtualbox.txt)" \
    --ostype "Linux26" \
    --memory "512" \
    --cpus "1" \
    --nic1 "intnet" \
    --intnet1 "port1" \
    --nic2 "intnet" \
    --intnet2 "port2" \
    --nic3 "intnet" \
    --intnet3 "port3" \
    --nic4 "nat" \
    --natpf4 "ssh,tcp,,2222,,22" \
    --natpf4 "luci,tcp,,8080,,80" \
    --uart1 "0x3F8" "4" \
    --uartmode1 "disconnected" && \
VBoxManage storagectl $VMNAME \
    --name "SATA Controller" \
    --add "sata" \
    --portcount "4" \
    --hostiocache "on" \
    --bootable "on" && \
VBoxManage storageattach $VMNAME \
    --storagectl "SATA Controller" \
    --port "1" \
    --type "hdd" \
    --nonrotational "on" \
    --medium $VDI

### Start VM with GUI
VBoxManage startvm $VMNAME --type "gui"

# wait until fully booted, hit enter. you will be root

vi /etc/config/network
# set the wan interface to eth3 and dhcp
config interface 'wan'
    option ifname   'eth3'
    option proto    'dhcp'

# restart network service
/etc/init.d/network restart

# now you should be able to access internet
ping www.google.com
# update package feed
opkg update
# set a root pwd
passwd

# append a rule to the firewall for allowing ssh
vi /etc/config/firewall
config rule
        option src              wan
        option proto            tcp
        option dest_port        22
        option target           ACCEPT

# restart the firewall
/etc/init.d/firewall restart

# now you should be able to connect from localhost like this:
ssh root@localhost -p 2222

VBoxManage controlvm $VMNAME poweroff
VBoxManage startvm $VMNAME --type headless

### Setup the 3-port switch

# replace the existing 'lan' section
vi /etc/config/network
# by
config interface 'lan1'
	option ifname 'eth0'
	option proto 'static'
	option ipaddr '192.168.11.1'
	option netmask '255.255.255.0'
	option ip6assign '60'

config interface 'lan2'
	option ifname 'eth1'
	option proto 'static'
	option ipaddr '192.168.12.1'
	option netmask '255.255.255.0'
	option ip6assign '60'

config interface 'lan3'
	option ifname 'eth2'
	option proto 'static'
	option ipaddr '192.168.13.1'
	option netmask '255.255.255.0'
	option ip6assign '60'

# restart network
/etc/init.d/network reload

### DHCP daemon reconfiguration
vi /etc/config/dhcp

# replace the existing 'lan' section
vi /etc/config/dhcp
# by
config dhcp 'lan1'
	option interface 'lan1'
	option start '100'
	option limit '150'
	option leasetime '12h'
	option dhcpv6 'server'
	option ra 'server'
	list 'dhcp_option' '3,192.168.11.1'

config dhcp 'lan2'
	option interface 'lan2'
	option start '100'
	option limit '150'
	option leasetime '12h'
	option dhcpv6 'server'
	option ra 'server'
	list 'dhcp_option' '3,192.168.12.1'

config dhcp 'lan3'
	option interface 'lan3'
	option start '100'
	option limit '150'
	option leasetime '12h'
	option dhcpv6 'server'
	option ra 'server'
	list 'dhcp_option' '3,192.168.13.1'

# restart the dhcp and dnsmasq services
/etc/init.d/odhcpd reload
/etc/init.d/dnsmasq reload

# replace the existing 'lan' zone
vi /etc/config/firewall
# by
config zone
	option name             lan
	list   network          'lan1'
	list   network          'lan2'
	list   network          'lan3'
	option input            ACCEPT
	option output           ACCEPT
	option forward          ACCEPT

# restart the firewall
/etc/init.d/firewall reload

# Now we are ready to create clients and attach them to the 3-switch port
